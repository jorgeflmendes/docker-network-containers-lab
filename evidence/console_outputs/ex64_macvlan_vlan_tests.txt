===== ub1 macvlan/VLAN setup =====
set +e
for c in b1 b2 r1 r2; do docker rm -f $c 2>/dev/null; done
docker network rm bluevlan redvlan 2>/dev/null
ip link del ens3.10 2>/dev/null
ip link del ens3.20 2>/dev/null
set -e
ip link add link ens3 name ens3.10 type vlan id 10
ip link add link ens3 name ens3.20 type vlan id 20
ip link set ens3.10 up
ip link set ens3.20 up
docker network create -d macvlan --subnet=10.60.10.0/24 -o parent=ens3.10 bluevlan
docker network create -d macvlan --subnet=10.60.20.0/24 -o parent=ens3.20 redvlan

ip -d link show ens3.20 | head -3
