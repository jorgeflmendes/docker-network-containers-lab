===== ub1 pre-swarm reset =====
set +e
for c in b1 b2 b3 b4 b5 b6 r1 r2 r3 r4; do docker rm -f $c 2>/dev/null; done
docker service rm bluesvc redsvc 2>/dev/null
docker network rm bluenet rednet bluevlan redvlan blue red 2>/dev/null
ip link del blue 2>/dev/null
ip link del red 2>/dev/null
ip link del ens3.10 2>/dev/null
ip link del ens3.20 2>/dev/null
docker swarm leave --force 2>/dev/null
iptables -P FORWARD ACCEPT
docker info --format 'node={{.Name}} swarm={{.Swarm.LocalNodeState}}'
