===== ub1 VXLAN setup =====
set +e
for c in b1 b2 b3 b4 b5 b6 r1 r2 r3 r4; do docker rm -f $c 2>/dev/null; done
docker network rm bluenet rednet bluevlan redvlan 2>/dev/null
ip link del blue 2>/dev/null
ip link del red 2>/dev/null
ip link del ens3.10 2>/dev/null
ip link del ens3.20 2>/dev/null
iptables -P FORWARD ACCEPT
set -e
ip link add blue type vxlan id 33 group 239.1.1.3 ttl 5 dev ens3 dstport 4789
ip link add red type vxlan id 66 group 239.1.1.6 ttl 5 dev ens3 dstport 4789
ip link set blue up
ip link set red up
ip addr add 10.0.0.101/24 dev blue
ip addr add 11.0.0.101/24 dev red
docker network create -d macvlan --subnet=10.0.0.0/24 --gateway=10.0.0.101 -o parent=blue bluenet
docker network create -d macvlan --subnet=11.0.0.0/24 --gateway=11.0.0.101 -o parent=red rednet
docker run -dit --name b1 --network bluenet --ip 10.0.0.1 alpine sleep 3600
docker run -dit --name b2 --network bluenet --ip 10.0.0.2 alpine sleep 3600
docker run -dit --name r1 --network rednet --ip 11.0.0.1 alpine sleep 3600
docker run -dit --name r2 --network rednet --ip 11.0.0.2 alpine sleep 3600
ip -d link show blue | grep vxlan
ip -d link show red | grep vxlan
